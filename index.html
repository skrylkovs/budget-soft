App.jsx
import { useState } from 'react'
import { Switch } from '@headlessui/react'
import { saveSettings } from '@/api/settings'

export function PreferencesForm({ initial }) {
  const [marketing, setMarketing] = useState(initial.marketing)
  const [updates, setUpdates] = useState(initial.updates ?? true)

  async function onSubmit(e) {
    e.preventDefault()
    await saveSettings({ marketing, updates })
  }

  return (
    <form onSubmit={onSubmit}>
      <Switch checked={marketing} onChange={setMarketing} />
      <Switch checked={updates} onChange={setUpdates} />
      <button type="submit">Сохранить</button>
    </form>
  )
}
